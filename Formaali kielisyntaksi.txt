<käännösyksikkö> ::= { <asetuslause> | <evaluaatio> | <toistosilmukka> | <tulostus> | päätä_ohjelma } erotin
                        | { <asetuslause> | <evaluaatio> | <toistosilmukka> | <tulostus> | päätä_ohjelma } erotin <käännösyksikkö>

<asetuslause> ::= muuttujanimi asetus <muuttujanarvo>
                    | muuttujanimi asetus muuttujanimi plus muuttujanimi
                    | muuttujanimi asetus muuttujanimi miinus muuttujanimi
                    | muuttujanimi asetus muuttujanimi kertaa muuttujanimi
                    | muuttujanimi asetus muuttujanimi jaettuna muuttujanimi

<muuttujanarvo> ::= luku
                    | merkkijono
                    | totuusarvo

<evaluaatio> ::= jos <evaluaatiolauseke> niin <käännösyksikkö>
                | jos <evaluaatiolauseke> tai <evaluaatio>
                | jos <evaluaatiolauseke> ja <evaluaatio>
                | ei <evaluaatio>


<evaluaatiolauseke> ::= evaluaatio_auki muuttujanimi suurempi_kuin muuttujanimi evaluaatio_kiinni
                        | evaluaatio_auki muuttujanimi pienempi_kuin muuttujanimi evaluaatio_kiinni
                        | evaluaatio_auki muuttujanimi yhtäsuuri_kuin muuttujanimi evaluaatio_kiinni

<toistosilmukka> ::= niinkauankuin <evaluaatio> seuraus <käännösyksikkö> päätä_silmukka

<tulostus> ::= tulosta_päätteeseen tulosta_auki muuttujanimi tulosta_kiinni
